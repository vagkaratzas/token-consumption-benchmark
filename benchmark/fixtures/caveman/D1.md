```python
# taskflow/models/task.py
from datetime import datetime, timezone
from typing import Optional

@dataclass
class Task(BaseModel):
    title: str = ""
    description: str = ""
    status: Status = Status.TODO
    project_id: Optional[int] = None
    assignee_id: Optional[int] = None
    due_date: Optional[str] = None  # ISO-8601 date string

    @property
    def is_overdue(self) -> bool:
        if self.due_date is None:
            return False
        try:
            due = datetime.fromisoformat(self.due_date)
        except ValueError:
            return False
        return due < datetime.now(timezone.utc) and not self.is_done()
```

```python
# taskflow/services/task_service.py
    def create_task(
        self, title: str, project_id: int, assignee_id: Optional[int] = None,
        due_date: Optional[str] = None,
    ) -> Task:
        if due_date is not None:
            try:
                datetime.fromisoformat(due_date)
            except ValueError:
                raise ValidationError(f"invalid due_date: {due_date!r}")
        task = Task(title=title, project_id=project_id,
                    assignee_id=assignee_id, due_date=due_date)
        task = self.tasks.add(task)
        if assignee_id is not None:
            self.notifications.notify(assignee_id, f"Assigned task {task.id}")
        return task

    def overdue_tasks(self, project_id: int) -> List[Task]:
        return self.tasks.list_overdue(project_id)
```

```python
# taskflow/repositories/task_repo.py
    def list_overdue(self, project_id: int) -> List[Task]:
        return [t for t in self.list_by_project(project_id) if t.is_overdue]
```

```python
# taskflow/api/routes_tasks.py
    def create_task(app: App, body: dict) -> dict:
        due_date = body.get("due_date")
        if due_date is not None and not isinstance(due_date, str):
            return {"status": 400, "error": "due_date must be a string"}
        task = app.services.tasks.create_task(
            title=body.get("title", ""),
            project_id=body.get("project_id"),
            assignee_id=body.get("assignee_id"),
            due_date=due_date,
        )
        return {"status": 201, "task": task.to_dict()}
```

```python
def test_create_task_with_due_date():
    services = build_services()
    task = services.tasks.create_task(
        "Write report", project_id=1, assignee_id=7,
        due_date="2099-01-01T00:00:00+00:00",
    )
    assert task.due_date == "2099-01-01T00:00:00+00:00"
    assert task.to_dict()["due_date"] == "2099-01-01T00:00:00+00:00"
    assert task.is_overdue is False
```
